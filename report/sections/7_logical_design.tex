    \item \textbf{EDIZIONE}.CodiceInsegnamento $\rightarrow$ \textbf{INSEGNAMENTO}.CodiceInsegnamento

    \item \textbf{ESAME}.Matricola $\rightarrow$ \textbf{STUDENTE}.Matricola
    \item \textbf{ESAME}.CodiceInsegnamento $\rightarrow$ \textbf{INSEGNAMENTO}.CodiceInsegnamento
    \item \textbf{PIANO\_DI\_STUDIO}.Matricola $\rightarrow$ \textbf{STUDENTE}.Matricola
    \item \textbf{PIANO\_DI\_STUDIO}.CodiceInsegnamento $\rightarrow$ \textbf{INSEGNAMENTO}.CodiceInsegnamento
    \item \textbf{PREREQUISITO}.CodiceInsegnamentoPropedeutico $\rightarrow$ \textbf{INSEGNAMENTO}.CodiceInsegnamento
    \item \textbf{PREREQUISITO}.CodiceInsegnamentoSuccessivo $\rightarrow$ \textbf{INSEGNAMENTO}.CodiceInsegnamento
    \item \textbf{PUO\_INSEGNARE}.CodiceFiscale $\rightarrow$ \textbf{DOCENTE}.CodiceFiscale
    \item \textbf{PUO\_INSEGNARE}.CodiceInsegnamento $\rightarrow$ \textbf{INSEGNAMENTO}.CodiceInsegnamento
    \item \textbf{TIENE}.CodiceFiscale $\rightarrow$ \textbf{DOCENTE}.CodiceFiscale
    \item \textbf{TIENE}.(CodiceInsegnamento, AnnoAccademico) $\rightarrow$ \textbf{EDIZIONE}.(CodiceInsegnamento, AnnoAccademico)
\end{itemize}
